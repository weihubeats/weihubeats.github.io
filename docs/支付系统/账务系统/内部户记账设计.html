内部户记账设计

表结构设计​

1. 账户交易记录表 - 宏观业务层​

记录一笔完整的业务动作。它站在业务系统的视角，只关心“发生了一笔什么交易

create table internal_account_transaction
(
    id               bigint auto_increment comment 'ID'
        primary key,
    trans_unique_id  varchar(64)                        not null comment '交易唯一ID',
    trans_type       varchar(32)                        not null comment '交易类型',
    trans_time       datetime                           not null comment '交易时间',
    biz_type         int                                not null comment '业务类型',
    biz_action       int                                not null comment '业务事件',
    biz_sn           varchar(64)                        not null comment '业务单号',
    sub_biz_sn       varchar(64)                        null comment '子业务单号',
    ori_biz_sn       varchar(64)                        null comment '原业务单号',
    description      varchar(128)                       null comment '交易摘要',
    status           varchar(16)                        not null comment '状态',
    is_reversed      varchar(1)                         not null comment '是否被冲正（Y/N）',
    ori_trans_id     bigint                             null comment '原交易ID',
    payment_channel  varchar(32)                        null comment '支付渠道',
    clearing_channel varchar(32)                        null comment '清算渠道',
    operator         varchar(32)                        not null comment '操作者',
    ext_info         varchar(1024)                      null comment '扩展信息',
    accounting_date  varchar(8)                         null comment '会计日',
    create_time      datetime default CURRENT_TIMESTAMP not null comment '创建时间',
    update_time      datetime default CURRENT_TIMESTAMP not null on update CURRENT_TIMESTAMP comment '更新时间',
    constraint uk_trans_unique_id
        unique (trans_unique_id)
)
    comment '账户交易记录';

create index idx_biz_sn_sub_biz_sn
    on internal_account_transaction (biz_sn, sub_biz_sn);

2. 账户交易分录表 - 中观规则层​

将一笔业务交易拆解为多个会计记账指令（即会计分录）。在复式记账法下，“有借必有贷，借贷必相等”。一笔交易通常会涉及两个或多个账户（例如：钱从商户扣除，加给用户）

create table internal_account_transaction_entry
(
    id                      bigint auto_increment comment 'ID'
        primary key,
    trans_id                bigint                             not null comment '交易ID',
    ori_trans_entry_id      bigint                             null comment '原交易分录ID',
    account_id              bigint                             null comment '账户ID',
    counterparty_id         bigint                             null comment '交易对手方ID',
    counterparty_account_id bigint                             null comment '交易对手方帐户ID',
    trans_type              varchar(32)                        not null comment '交易类型',
    trans_time              datetime                           not null comment '交易时间',
    variety_code            varchar(20)                        not null comment '币种',
    amount                  decimal(20, 8)                     not null comment '交易金额',
    action                  varchar(16)                        not null comment '交易动作',
    freeze_type             varchar(32)                        null comment '冻结类型',
    fee_item                varchar(32)                        null comment '费用项',
    ext_info                varchar(1024)                      null comment '扩展信息',
    accounting_entity       varchar(32)                        null comment '核算主体',
    accounting_date         varchar(8)                         null comment '会计日',
    create_time             datetime default CURRENT_TIMESTAMP not null comment '创建时间',
    update_time             datetime default CURRENT_TIMESTAMP not null on update CURRENT_TIMESTAMP comment '更新时间',
    status                  varchar(32)                        null comment '状态',
    task_id                 bigint   default -1                not null comment '缓冲记账任务id'
)
    comment '账户交易条目';

create index idx_account_id_trans_time
    on internal_account_transaction_entry (account_id, trans_time);

create index idx_task_id_status_account_id
    on internal_account_transaction_entry (task_id, status, account_id);

3. customer_account_transaction_log 账户交易流水表 - 微观资金层​

记录账户资金的真实物理变动轨迹（快照）。这是最底层的账本，用于对账、审计和排查资金问题

create table internal_account_transaction_log
(
    id                      bigint auto_increment comment 'ID'
        primary key,
    account_id              bigint                             not null comment '账户ID',
    counterparty_id         bigint                             null comment '交易对手方ID',
    counterparty_account_id bigint                             null comment '交易对手方帐户ID',
    trans_entry_id          bigint                             not null comment '交易分录ID',
    ori_trans_entry_id      bigint                             null comment '原交易分录ID',
    trans_type              varchar(32)                        not null comment '交易类型',
    trans_time              datetime                           not null comment '交易时间',
    description             varchar(128)                       null comment '交易摘要',
    biz_type                int                                not null comment '业务类型',
    biz_action              int                                not null comment '业务事件',
    biz_sn                  varchar(64)                        not null comment '业务单号',
    sub_biz_sn              varchar(64)                        null comment '子业务单号',
    ori_biz_sn              varchar(64)                        null comment '原业务单号',
    variety_code            varchar(20)                        not null comment '币种',
    amount                  decimal(20, 8)                     not null comment '交易金额',
    balance                 decimal(20, 8)                     not null comment '余额',
    previous_balance        decimal(20, 8)                     not null comment '先前余额',
    dr_cr                   varchar(1)                         not null comment '借贷方向（借D，贷C）',
    in_out                  varchar(1)                         not null comment '收支方向（收入I，支出O）',
    payment_channel         varchar(32)                        null comment '支付渠道',
    clearing_channel        varchar(32)                        null comment '清算渠道',
    freeze_type             varchar(32)                        null comment '冻结类型',
    freeze                  decimal(20, 8)                     not null comment '冻结金额',
    available               decimal(20, 8)                     not null comment '可用余额',
    fee_item                varchar(32)                        null comment '费用项',
    is_reversed             varchar(1)                         not null comment '是否被冲正（Y/N）',
    operator                varchar(32)                        not null comment '操作者',
    ext_info                varchar(1024)                      null comment '扩展信息',
    accounting_entity       varchar(32)                        null comment '核算主体',
    accounting_date         varchar(8)                         null comment '会计日',
    create_time             datetime default CURRENT_TIMESTAMP not null comment '创建时间',
    update_time             datetime default CURRENT_TIMESTAMP not null on update CURRENT_TIMESTAMP comment '更新时间'
)
    comment '账户交易流水';

create index idx_account_id_trans_time
    on internal_account_transaction_log (account_id, trans_time);

create index idx_biz_sn_sub_biz_sn
    on internal_account_transaction_log (biz_sn, sub_biz_sn);

为什么需要三张表​

  1. 解决“一对多”的复杂资金流转问题（复式记账的需求）

  • 问题：一笔简单的“退款”，并不是改一个数字那么简单。它可能涉及：扣减商户待结算户余额、增加用户可用余额、退还平台手续费等。

  • 解决：通过 Transaction (1) -> Entry (N) 的设计，一笔业务交易可以挂载无数个分录动作，完美支持复杂的资金拆分和多方记账。

  1. 业务逻辑与底层账务的解耦

  • 问题：上游业务系统（如电商订单系统）不懂什么是“借/贷 (dr_cr)”，也不关心底层的 account_id 是多少

  • 解决：Transaction 表对接上游业务（认 biz_sn），而 Log 表对接底层账户系统。Entry 充当翻译官，将业务动作转化为具体的资金拨拨指令。

  1. 保证账务的幂等性与可追溯性（防资损）

  • 问题：网络超时导致上游重试退款，如何防止给用户退两次钱？资金出错了怎么查？

  • 解决：Transaction 表的 trans_unique_id 配合 biz_sn 做唯一索引防重发（幂等）。Log 表记录了每一笔变动前后的 previous_balance 和 balance，形成了一条严密的资金链条（资金动帐轨迹），审计查账时一目了然。

举例说明​

用户买了一件 100 元的商品，平台抽成 2 元，商户实际到账 98 元。 之后发生了全额退款

期初状态（初始账户余额）

  • 买家账户 (ID: 1001, 负债类)：初始余额 500.00

  • 商户账户 (ID: 2001, 负债类)：初始余额 1000.00

  • 平台收入户 (ID: 3001, 权益类)：初始余额 50.00

正向记账​

业务动作：用户发起支付，单号 PAY20260310001。 账务拆解：

  1. 买家扣款 100.00

  2. 商户入账 98.00

  3. 平台收手续费 2.00

1. 记录业务交易 (Transaction)​

业务系统告诉账务系统：有一笔支付订单来了

INSERT INTO internal_account_transaction 
(id, trans_unique_id, trans_type, trans_time, biz_type, biz_action, biz_sn, status, is_reversed, operator) 
VALUES 
(1, 'TXN_PAY_001', 'PAYMENT', '2026-03-10 10:00:00', 1, 10, 'PAY20260310001', 'SUCCESS', 'N', 'SYSTEM');

2. 生成会计分录 (Entry)​

把这 1 笔交易拆成 3 个具体的记账指令

INSERT INTO internal_account_transaction_entry 
(id, user_id, mid, trans_id, trans_type, trans_time, account_id, account_sub_type, variety_code, amount, action) 
VALUES 
-- 指令1：买家付款
(101, 888, 1, 1, 'PAYMENT', '2026-03-10 10:00:01', 1001, 'USER_BALANCE', 'CNY', 100.00, 'PAY'),
-- 指令2：商户收款
(102, 999, 1, 1, 'PAYMENT', '2026-03-10 10:00:01', 2001, 'MERCHANT_SETTLE', 'CNY', 98.00, 'RECEIVE'),
-- 指令3：平台收手续费
(103, 0, 1, 1, 'PAYMENT', '2026-03-10 10:00:01', 3001, 'PLATFORM_INCOME', 'CNY', 2.00, 'FEE_IN');

3. 落地资金流水 (Log) - 核心账本​

INSERT INTO internal_account_transaction_log 
(id, trans_log_id, user_id, mid, account_id, account_sub_type, trans_id, trans_entry_id, trans_type, trans_time, amount, biz_type, biz_action, biz_sn, variety_code, balance, previous_balance, available, dr_cr, in_out, is_reversed, operator) 
VALUES 
-- 流水1：买家扣100 (期初500 -> 期末400)。买家负债减少记借(D)，余额减少记支出(O)
(1001, 80001, 888, 1, 1001, 'USER_BALANCE', 1, 101, 'PAYMENT', '2026-03-10 10:00:02', 100.00, 1, 10, 'PAY20260310001', 'CNY', 400.00, 500.00, 400.00, 'D', 'O', 'N', 'SYSTEM'),

-- 流水2：商户加98 (期初1000 -> 期末1098)。商户负债增加记贷(C)，余额增加记收入(I)
(1002, 80002, 999, 1, 2001, 'MERCHANT_SETTLE', 1, 102, 'PAYMENT', '2026-03-10 10:00:02', 98.00, 1, 10, 'PAY20260310001', 'CNY', 1098.00, 1000.00, 1098.00, 'C', 'I', 'N', 'SYSTEM'),

-- 流水3：平台加2 (期初50 -> 期末52)。平台收入增加记贷(C)，余额增加记收入(I)
(1003, 80003, 0, 1, 3001, 'PLATFORM_INCOME', 1, 103, 'PAYMENT', '2026-03-10 10:00:02', 2.00, 1, 10, 'PAY20260310001', 'CNY', 52.00, 50.00, 52.00, 'C', 'I', 'N', 'SYSTEM');

💡 验证会计恒等式：借方总额 (100) = 贷方总额 (98 + 2)，账平了！

逆向记账（退款/冲正场景）​

业务动作：用户对上面那笔订单发起了全额退款，单号 REF20260310002。 账务拆解（原路退回）：

  1. 商户账户扣回 98.00

  2. 平台账户扣回手续费 2.00

  3. 买家账户增加 100.00

  4. 记录业务交易 (Transaction)

1. 记录业务交易 (Transaction)​

注意这里的 ori_biz_sn（原业务单号），它把退款和原支付单绑定在了一起

INSERT INTO internal_account_transaction 
(id, trans_unique_id, trans_type, trans_time, biz_type, biz_action, biz_sn, ori_biz_sn, status, is_reversed, operator) 
VALUES 
(2, 'TXN_REF_001', 'REFUND', '2026-03-10 15:30:00', 2, 20, 'REF20260310002', 'PAY20260310001', 'SUCCESS', 'N', 'SYSTEM');

2. 生成会计分录 (Entry)​

INSERT INTO internal_account_transaction_entry 
(id, user_id, mid, trans_id, trans_type, trans_time, account_id, account_sub_type, variety_code, amount, action) 
VALUES 
-- 指令1：商户退款扣减
(201, 999, 1, 2, 'REFUND', '2026-03-10 15:30:01', 2001, 'MERCHANT_SETTLE', 'CNY', 98.00, 'REFUND_DEDUCT'),
-- 指令2：平台退还手续费
(202, 0, 1, 2, 'REFUND', '2026-03-10 15:30:01', 3001, 'PLATFORM_INCOME', 'CNY', 2.00, 'REFUND_FEE'),
-- 指令3：买家收到退款
(203, 888, 1, 2, 'REFUND', '2026-03-10 15:30:01', 1001, 'USER_BALANCE', 'CNY', 100.00, 'REFUND_ADD');
  1. 落地资金流水 (Log) - 核心账本

基于当前的余额（也就是支付完成后的余额），进行反向操作。

INSERT INTO internal_account_transaction_log 
(id, trans_log_id, user_id, mid, account_id, account_sub_type, trans_id, trans_entry_id, trans_type, trans_time, amount, biz_type, biz_action, biz_sn, variety_code, balance, previous_balance, available, dr_cr, in_out, is_reversed, operator) 
VALUES 
-- 流水1：商户扣98 (期初1098 -> 期末1000)。商户负债减少记借(D)，余额减少记支出(O)
(2001, 80004, 999, 1, 2001, 'MERCHANT_SETTLE', 2, 201, 'REFUND', '2026-03-10 15:30:02', 98.00, 2, 20, 'REF20260310002', 'CNY', 1000.00, 1098.00, 1000.00, 'D', 'O', 'N', 'SYSTEM'),

-- 流水2：平台扣2 (期初52 -> 期末50)。平台收入减少记借(D)，余额减少记支出(O)
(2002, 80005, 0, 1, 3001, 'PLATFORM_INCOME', 2, 202, 'REFUND', '2026-03-10 15:30:02', 2.00, 2, 20, 'REF20260310002', 'CNY', 50.00, 52.00, 50.00, 'D', 'O', 'N', 'SYSTEM'),

-- 流水3：买家加100 (期初400 -> 期末500)。买家负债增加记贷(C)，余额增加记收入(I)
(2003, 80006, 888, 1, 1001, 'USER_BALANCE', 2, 203, 'REFUND', '2026-03-10 15:30:02', 100.00, 2, 20, 'REF20260310002', 'CNY', 500.00, 400.00, 500.00, 'C', 'I', 'N', 'SYSTEM');

💡 再次验证：借方总额 (98+2) = 贷方总额 (100)，账依然是平的。经过这一正一反，所有账户余额都完美回到了原点
